"Error creating request %v"

response
,
 
err
 
:=
 
client
.
Do
(

if

nil
 
{

.
Errorf
(
communicationErrorWithServerErrorMessage
,

if

.

.
StatusOK

StatusCreated

body

ioutil
.
ReadAll
(
response

(

{

.
Errorf

)

"ok"

.
Errorf

"Error response: %s"

.
Status
)

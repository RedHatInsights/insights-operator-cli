tryToFindProfile
(
t
 
*
testing
.
T
,
 
captured
,

)
 
{

!
strings
.

(
captured
,
 
profileDescription
)
 
{

t
.
Fatal
(
"Can not find profile with description:"
,
 
profileDescription
)

	
}


}
